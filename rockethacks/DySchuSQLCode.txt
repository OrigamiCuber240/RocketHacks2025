CREATE DATABASE DySchu;

CREATE TABLE client(
	clientNum INT PRIMARY KEY,
	clientFirstName CHAR,
	clientSirName CHAR,
);

CREATE TABLE employee(
	employeeNum INT PRIMARY KEY,
	employeeFirstName CHAR,
	employeeName CHAR
);

CREATE TABLE clientEmployee(
	employeeNum INT,
	clientNum INT,
	FOREIGN KEY (employeeNum) REFERENCES employee (employeeNum)
	FOREIGN KEY (clientNum) REFERENCES employee (clientNum)
	clientEmployeeNum INT PRIMARY KEY
);
